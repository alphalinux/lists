#include <stdio.h>
#include <unistd.h>

int main(void)
{
        unsigned long int *p = sbrk(0);

        do {
                printf("%p\n", p);
                *p = 0xdeadbeefdeadbeef;
        } while(*p++);

        return 0;
}

$ cc -o coredump coredump.c
$ ulimit -c 16384
$ uname -a
Linux cassiel 2.2.1 #1 Mon Feb 1 10:54:51 NZDT 1999 alpha unknown
$ ./coredump

$ not a dynamic executable
