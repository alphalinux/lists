2.1.132 doesn't have the SMP line in the Makefile. That's now been moved to
a configuration option.
The problem appears to be in some changes which didn't make it into the
asm-alpha/softirq.h file.

This patch works here. It just transfers the i386 changes to the alpha
routines.

--- include/asm-alpha/softirq.h.orig Sat Dec 26 00:51:33 1998
+++ include/asm-alpha/softirq.h Sat Dec 26 00:55:03 1998
@@ -27,7 +27,8 @@
 extern inline void init_bh(int nr, void (*routine)(void))
 {
        bh_base[nr] = routine;
- bh_mask_count[nr] = 0;
+ /* bh_mask_count[nr] = 0; */
+ atomic_set(&bh_mask_count[nr], 0);
        bh_mask |= 1 << nr;
 }
 
@@ -116,12 +117,14 @@
 extern inline void disable_bh(int nr)
 {
        bh_mask &= ~(1 << nr);
- bh_mask_count[nr]++;
+ /* bh_mask_count[nr]++; */
+ atomic_inc(&bh_mask_count[nr]);
 }
 
 extern inline void enable_bh(int nr)
 {
- if (!--bh_mask_count[nr])
+ /* if (!--bh_mask_count[nr]) */
+ if (atomic_dec_and_test(&bh_mask_count[nr]))
                bh_mask |= 1 << nr;
 }
