here is that patch:

Make sure the following replaces the appropriate stuff at the end of the
file: include/asm-alpha/a.out.h

#ifdef __KERNEL__ #define STACK_TOP ((current->personality &
ADDR_MAX_32BIT) ? 0x100000000UL : \
((current->personality&ADDR_MAX_31BIT) ? 0x80000000UL : 0x00120000000UL))
#endif
