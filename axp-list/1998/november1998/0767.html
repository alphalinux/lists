Just to follow me up :)

Here is my favorite mm subsystem tester. Have fun and don't wonder about
`cannot load interpreter' messages. ;)

#define SIZE (2 << 20)

main() {
        char m[SIZE + 1];
        int i;

        while(fork() != -1)
                for(i = 0; i < SIZE; i++)
                        m[i] = m[i] + i & 0xff;
        perror("fork");
}
