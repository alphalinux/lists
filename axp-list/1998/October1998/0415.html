Just before reading the mails I found some time to have a look at the PGP
sources and fixed the problem. None-the-less, thanks to all for your
help; I now know that my patch was correct.

Here's the patch I used:

diff -ur pgp50i.old/src/lib/pgp/include/pgpUsuals.h pgp50i/src/lib/pgp/include/pgpUsuals.h
--- pgp50i.old/src/lib/pgp/include/pgpUsuals.h Thu Nov 19 18:06:12 1998
+++ pgp50i/src/lib/pgp/include/pgpUsuals.h Thu Nov 19 18:08:04 1998
@@ -66,13 +66,13 @@
 * IRIX uses ULONGLONG_MAX. Are there any other names for this?
 */
 
-#ifndef HAVE64
-#if defined(ULONG_LONG_MAX) || defined (ULLONG_MAX) || defined(ULONGLONG_MAX)
+ //#ifndef HAVE64
+ //#if defined(ULONG_LONG_MAX) || defined (ULLONG_MAX) || defined(ULONGLONG_MAX)
 typedef unsigned long long word64;
 typedef long long int64;
 #define HAVE64 1
-#endif
-#endif
+ //#endif
+ //#endif
