I had that prob when I went to kernel 2.1.x. You need ppp-2.3.3.
Here is my ifcfg-ppp0:

PERSIST=yes
DEFROUTE=yes
ONBOOT=yes
INITSTRING='AT &F2 Z'
MODEMPORT=/dev/modem
LINESPEED=38400
ESCAPECHARS=no
DEFABORT=yes
HARDFLOWCTL=yes
DEVICE=ppp0
PPPOPTIONS=
DEBUG=yes
PAPNAME=
REMIP=0.0.0.0
IPADDR=0.0.0.0
BOOTPROTO=none
MTU=
MRU=
DISCONNECTTIMEOUT=
RETRYTIMEOUT=
USERCTL=no
