test program. It works well on our Solaris machines, but fails with an
"Operation not permitted" error on my Alpha running RH Linux 5.0, both
using stdin and file pointers. The RH errata makes no mention of any
problems. Can anyone be of assistance?

#include <stdio.h>
#include <poll.h>


main()
{
	struct pollfd fds[1];
	int n;
	char buffer[BUFSIZ];
/*
	FILE *fp;


	if (NULL == (fp = fopen ("poll_test.tmp", "rw"))) {
		perror ("fopen");
	}


	fds[0].fd = fileno (fp);
*/
	fds[0].fd = fileno (stdin);
	fds[0].events = POLLIN;


	while (1) {
		sleep (1);


		n = poll (fds, (unsigned long int) 1, 2000);
		if (-1 == n) perror ("poll");
		printf ("%d events = %X\trevents = %X\n",
			n, fds[0].events, fds[0].revents);


		if (POLLIN == fds[0].revents) {
			printf ("Found input event\n");
			fgets (buffer, BUFSIZ, stdin);
		}
	}
}
