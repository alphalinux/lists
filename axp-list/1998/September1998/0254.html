> do anybody know where can i get a pgp5.0i binary distribution?. I'm having
> troubles to compile the src package on my system... :m

Here's a patch:

diff -ur pgp50i.orig/src/lib/pgp/hash/pgpMD5.c pgp50i/src/lib/pgp/hash/pgpMD5.c
--- pgp50i.orig/src/lib/pgp/hash/pgpMD5.c Sat Aug 9 22:44:50 1997
+++ pgp50i/src/lib/pgp/hash/pgpMD5.c Mon Sep 14 19:53:31 1998
@@ -39,7 +39,7 @@
word32 key[MD5_BLOCKWORDS];
word32 iv[MD5_HASHWORDS];
#if HAVE64
- word64 bytes;
+ bnword64 bytes;
#else
word32 bytesLo, bytesHi;
#endif
diff -ur pgp50i.orig/src/lib/pgp/hash/pgpSHA.c pgp50i/src/lib/pgp/hash/pgpSHA.c
--- pgp50i.orig/src/lib/pgp/hash/pgpSHA.c Sat Aug 9 22:44:51 1997
+++ pgp50i/src/lib/pgp/hash/pgpSHA.c Mon Sep 14 19:53:20 1998
@@ -46,7 +46,7 @@
word32 key[SHA_BLOCKWORDS];
word32 iv[SHA_HASHWORDS];
#if HAVE64
- word64 bytes;
+ bnword64 bytes;
#else
word32 bytesHi, bytesLo;
#endif
diff -ur pgp50i.orig/src/lib/pgp/random/pgpRndPool.c pgp50i/src/lib/pgp/random/pgpRndPool.c
--- pgp50i.orig/src/lib/pgp/random/pgpRndPool.c Sat Aug 9 22:45:13 1997
+++ pgp50i/src/lib/pgp/random/pgpRndPool.c Mon Sep 14 19:53:53 1998
@@ -300,7 +300,7 @@
/* Inline asm goodies */
#define UMULH_32(r,a,b) __asm__("mull %2" : "=d"(r) : "%a"(a), "mr"(b) : "ax")
#elif HAVE64
-#define UMULH_32(r,a,b) ((r) = (word32)((word64)(a) * (b) >> 32))
+#define UMULH_32(r,a,b) ((r) = (word32)((bnword64)(a) * (b) >> 32))
#else
/* Underestimate the product */
#define UMULH_32(r,a,b) ((r) = ((a) >> 16) * ((b) >> 16))
