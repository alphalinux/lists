following patch should clear up the undefined references:

diff -ruN pcmcia-cs-3.0.9/debug-tools/dump_i365.c pcmcia-cs-patched/debug-tools/dump_i365.c
--- pcmcia-cs-3.0.9/debug-tools/dump_i365.c Sun Jul 26 19:10:51 1998
+++ pcmcia-cs-patched/debug-tools/dump_i365.c Wed May 5 00:20:51 1999
@@ -41,8 +41,13 @@
 #include <asm/io.h>
 #endif
 #include <unistd.h>
+#ifndef __alpha__
 #define INB(a) inb_p(a)
 #define OUTB(d, a) outb_p(d, a)
+#else /* __alpha__ */
+#define INB(a) inb(a)
+#define OUTB(d, a) outb(d, a)
+#endif /* __alpha__ */
 
 #endif /* __MSDOS__ */
