It's much easier to build a generic alpha kernel and then just
"tweek" .../linux/arch/alpha/Makefile

  ifeq ($(CONFIG_ALPHA_GENERIC),y)
    CFLAGS := $(CFLAGS) -mcpu=ev6 <---------<<<<
  endif

to indicate your cpu type. (the Makefile is "shipped" with -mcpu=ev5")
