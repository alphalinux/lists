I want to convert assembly code from I386 to AXP,How do it?

for example,converting following code from I386 to AXP:

static void inline port_out(char value, unsigned short port)
{
__asm__ volatile ("outb ,"
                ::"a" ((char) value),"d" ((unsigned short) port));
}

static unsigned char inline port_in(unsigned short port)
{
        unsigned char _v;
__asm__ volatile ("inb ,"
                :"=a" (_v):"d" ((unsigned short) port));
        return _v;
}
