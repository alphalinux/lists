At 05:10 PM 10/26/1999 , Mark Holloway wrote:

Almost forgot!

For a file to be written to the directory, you have to create (touch) the
file in /tftpboot.

So, to store a file called routercfg, you would, as root, type...

[root@pen /tftpboot]# ls
total 569
-rwxrwxrwx 1 root root 104546 Feb 24 1999 david*
-rwxrwxrwx 1 root root 72302 Jan 19 1999 in-n-outr3100*
-rw-rw-r-- 1 root root 347 Mar 11 1999 listing
-rwxrwxrwx 1 root root 104694 Dec 13 1998 netcfg*
-rwxrwxrwx 1 root root 72302 Jan 6 1999 ntconfinbnd*
-rwxrwxrwx 1 root root 72302 Mar 11 1999 r3100-3.11.99*
-rwxrwxrwx 1 root root 72302 Feb 2 1999 r3100cfg*
-rwxrwxrwx 1 root root 72302 Mar 11 1999 r3100ortinat*
[root@pen /tftpboot]# touch routercfg
[root@pen /tftpboot]# ls
total 569
-rwxrwxrwx 1 root root 104546 Feb 24 1999 david*
-rwxrwxrwx 1 root root 72302 Jan 19 1999 in-n-outr3100*
-rw-rw-r-- 1 root root 347 Mar 11 1999 listing
-rwxrwxrwx 1 root root 104694 Dec 13 1998 netcfg*
-rwxrwxrwx 1 root root 72302 Jan 6 1999 ntconfinbnd*
-rwxrwxrwx 1 root root 72302 Mar 11 1999 r3100-3.11.99*
-rwxrwxrwx 1 root root 72302 Feb 2 1999 r3100cfg*
-rwxrwxrwx 1 root root 72302 Mar 11 1999 r3100ortinat*
-rw-rw-r-- 1 root root 0 Oct 26 17:25 routercfg
[root@pen /tftpboot]# chmod a+rwx routercfg
[root@pen /tftpboot]# ls
total 569
-rwxrwxrwx 1 root root 104546 Feb 24 1999 david*
-rwxrwxrwx 1 root root 72302 Jan 19 1999 in-n-outr3100*
-rw-rw-r-- 1 root root 347 Mar 11 1999 listing
-rwxrwxrwx 1 root root 104694 Dec 13 1998 netcfg*
-rwxrwxrwx 1 root root 72302 Jan 6 1999 ntconfinbnd*
-rwxrwxrwx 1 root root 72302 Mar 11 1999 r3100-3.11.99*
-rwxrwxrwx 1 root root 72302 Feb 2 1999 r3100cfg*
-rwxrwxrwx 1 root root 72302 Mar 11 1999 r3100ortinat*
-rwxrwxrwx 1 root root 0 Oct 26 17:25 routercfg*
[root@pen /tftpboot]#
