Alvin:

There was a uac utility with Digital Unix that could control how unaligned
accesses were handled. Here's a little program that spawns a shell with the
fixup's disabled. When you run your program it will halt instead of doing
the fixups and you can then eliminate them. Note this can be "tedious" and
some programs such as emacs spew out unaligned accesses so be careful what
you run. Note also this code is old and I haven't used it in a while so it
may need a bit of rework but gives a place to start.

Frank

--------------------------------------------------

#include <asm/sysinfo.h>
#include <asm/unistd.h>

static int setsysinfo( unsigned long op, void *buffer, unsigned long size,
int *
start, void *arg, unsigned long flag )
{
  syscall( __NR_osf_setsysinfo, op, buffer, size, start, arg, flag );
}

void trap_unaligned(void)
{
  unsigned int buf[2];
  buf[0] = SSIN_UACPROC;
  buf[1] = UAC_SIGBUS | UAC_NOPRINT;
  setsysinfo( SSI_NVPAIRS, buf, 1, 0, 0, 0 );
}

int main( int ac )
{
  trap_unaligned();
  printf("Creating a subshell with SIGBUS setting!\n" );
  execl("/bin/bash", "/bin/bash", (char *)0);
  return(0);
}
