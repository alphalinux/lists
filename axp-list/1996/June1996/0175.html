Erik Troan suggest to use /proc/cpuinfo to identify a Jensen
automatically. Insert the following code before the call to cmos_init
to accomplish this:

if (clockport != 0x170) { /* use jensen!=0x100 if you use my
original patch */
FILE *cpuinfo;
char field[80];
if(cpuinfo = fopen("/proc/cpuinfo","r")){
while(!feof(cpuinfo)){
if(1==fscanf(cpuinfo,"system type : %s",field)){
if (0==strcmp(field,"Jensen"))
clockport = 0x170;
/* use jensen=0x100; if you use my original patch */
fclose(cpuinfo);
break;
}
fgets(field,80,cpuinfo);
}
fclose(cpuinfo);
}
}
