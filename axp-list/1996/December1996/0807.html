Here is the XF86Config file for my UDB using the latest TGA X server.
This file also works with the previous X servers that I had before.
Simply cut and past and it should work.

###
# **********************************************************************
#Do a "cat /proc/pci" and examine it for the DEC DC21030 (TGA) device.
#
#On an Alpha XL-266 it looks like:
#
#PCI devices found:
# Bus 0, device 13, function 0:
# Display controller: DEC DC21030 (rev 2).
# Medium devsel. Fast back-to-back capable. Master Capable. Latency=32.
# Prefetchable 32 bit memory at 0x2000000.
#
#
#and on the UDB, like:
#
#> PCI devices found:
#> Bus 0, device 11, function 0:
#> Display controller: DEC DC21030 (rev 2).
#> Medium devsel. Fast back-to-back capable. Master Capable. Latency=15.
#> Prefetchable 32 bit memory at 0x2000000.
# **********************************************************************
#

# **********************************************************************
# Files section. This allows default font and rgb paths to be set
# **********************************************************************

Section "Files"
RgbPath "/usr/lib/X11/rgb"

FontPath "/usr/lib/X11/fonts/misc/"
FontPath "/usr/lib/X11/fonts/75dpi/"
FontPath "/usr/lib/X11/fonts/100dpi/"
FontPath "/usr/lib/X11/fonts/Type1/"
FontPath "/usr/lib/X11/fonts/Speedo/"
EndSection

# **********************************************************************
# Input devices
# **********************************************************************

# **********************************************************************
# Keyboard section
# **********************************************************************

Section "Keyboard"
Protocol "Standard"
LeftAlt Meta
RightAlt ModeShift
# RightCtl Compose
# ScrollLock ModeLock
EndSection

# **********************************************************************
# Pointer section
# **********************************************************************

Section "Pointer"
Protocol "PS/2"
# Protocol "Microsoft"
# Device "/dev/mouse"
Device "/dev/psaux"
# Emulate3Buttons
# Emulate3Timeout 50
EndSection

# **********************************************************************
# Monitor section
# **********************************************************************

Section "Monitor"
Identifier "1024x768 Monitor"
HorizSync 31-50
VertRefresh 55-90

#Uncomment a mode line to use that resolution, only one mode line
#may be uncommetted at any one time. Make sure you change the resolution
#at the bottom of this file to match the mode line

#Modeline "1024x768" 75 1024 1048 1184 1328 768 771 777 806 -hsync -vsync
#640x480 72Hz 37.86kHz 32.0Mhz

#This mode is good for games (ie. MasterGear runs nice on the UDB!)
#Modeline "640x480" 32.00 640 692 796 848 480 491 513 524
#800x600 72Hz 48.08kHz 50.00Mhz

#The ideal resolution for running UAE Amiga Emulator on the UDB
#Modeline "800x600" 50.00 800 860 980 1040 600 616 651 667

# 1024x768 @ 60 Hz, 48.4 kHz hsync
Modeline "1024x768" 65 1024 1032 1176 1344 768 771 777 806 -hsync -vsync

#1024x768 72Hz 57.38kHz 74.37Mhz
#The highest res still at VESA for my sony monitor
#Modeline "1024x768" 74.37 1024 1092 1228 1296 768 775 790 797

#1024x864 60Hz 54..05kHz 69.00Mhz
#An odd resolution that TGA supports for old DEC monitors
#Modeline "1024x864" 69.00 1024 1088 1216 1280 864 872 890 898

#1152x900 66Hz 61.79kHz 92.98Mhz
#The standard SUN mode and probably the most best choice for the UDB
#Modeline "1152x900" 92.98 1152 1240 1416 1504 900 909 927 936

#1280x1024 60Hz 63.34kHz 108.18Mhz
#Highest res on the UDB but the lower refresh...my monitor can't do better
#Modeline "1280x1024" 108.18 1280 1388 1604 1712 1024 1031 1046 1053

EndSection

# **********************************************************************
# Graphics device section
# **********************************************************************
Section "Device"
Identifier "TGA"
VendorName "DEC"
BoardName "ZLXp-E1"
# NOTE: for now, MemBase option is *REQUIRED* !!!!!!
MemBase 0x02000000
# NOTE: for now, use the XFree86 software cursor support!!!!
Option "sw_cursor"
EndSection

# **********************************************************************
# Screen sections
# **********************************************************************

# The accelerated servers (S3, Mach32, Mach8, 8514, P9000, AGX, W32, Mach64)
# ...and now, DEC's TGA!!! :-)

Section "Screen"
Driver "accel"
Device "TGA"
Monitor "1024x768 Monitor"
Subsection "Display"
# Please note that the following line needs to be changed to the
# resolution you uncommetted above. i
# e.g. 800x600 if you uncommetted that mode
Modes "1024x768"
ViewPort 0 0
EndSubSection
EndSection
