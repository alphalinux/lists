below is the fix for the corrupted fonts e.g. in the "control-panel"
and a second small one.

-------------------------------------------------------------------------------
diff -u /mnt/new/x/xc/programs/Xserver/cfb/cfb8bit.h /soft/r6/xc/programs/Xserver/cfb/cfb8bit.h
--- /mnt/new/x/xc/programs/Xserver/cfb/cfb8bit.h Tue Jan 16 19:21:50 1996
+++ /soft/r6/xc/programs/Xserver/cfb/cfb8bit.h Sun May 5 22:07:49 1996
@@ -394,7 +394,7 @@
break; \
case 6: \
((CARD8 *) (dst))[SinglePixel5] = (pixel); \
- ((CARD8 *) (dst))[SinglePixel7] = (pixel); \
+ ((CARD8 *) (dst))[SinglePixel6] = (pixel); \
break; \
case 7: \
((CARD16 *) (dst))[DoublePixel2] = (pixel); \
--- /mnt/new/x/xc/programs/Xserver/dix/window.c Sat Apr 27 20:44:51 1996
+++ xc/programs/Xserver/dix/window.c Mon Apr 29 23:38:32 1996
@@ -1126,14 +1126,14 @@
pWin->forcedBS = FALSE;
break;
case CWBackingPlanes:
- if (pWin->optional || ((CARD32)*pVlist != ~0L)) {
+ if (pWin->optional || ((CARD32)*pVlist != (CARD32)~0L)) {
if (!pWin->optional && !MakeWindowOptional (pWin))
{
error = BadAlloc;
goto PatchUp;
}
pWin->optional->backingBitPlanes = (CARD32) *pVlist;
- if ((CARD32)*pVlist == ~0L)
+ if ((CARD32)*pVlist == (CARD32)~0L)
checkOptional = TRUE;
}
pVlist++;
