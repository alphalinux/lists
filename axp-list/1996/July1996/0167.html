Are you certain the patch applied cleanly? The best way is to cd
into the root directory of the kernel sources, then do
"zcat patchfile.gz | patch -p1". The timespecbits.h file should
be there. This is from axp-diffs-2.0.7.gz:

diff -urN 386/include/linux/timespecbits.h linux/include/linux/timespecbits.h
--- 386/include/linux/timespecbits.h Wed Dec 31 17:00:00 1969
+++ linux/include/linux/timespecbits.h Fri Jun 14 10:48:02 1996
@@ -0,0 +1,14 @@
+#ifndef _LINUX_TIMESPECBITS_H
+#define _LINUX_TIMESPECBITS_H
+
+struct timespec {
+ long tv_sec; /* seconds */
+ long tv_nsec; /* nanoseconds */
+};
+
+struct itimerspec {
+ struct timespec it_interval; /* timer period */
+ struct timespec it_value; /* timer expiration */
+};
+
+#endif /* _LINUX_TIMESPECBITS_H */
