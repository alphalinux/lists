Enclosed is a patch to ipfwadm.c to deal with system
clocks not running at 100Hz.

r~

--- ipfwadm.c.orig Mon Jun 17 05:33:11 1996
+++ ipfwadm.c Thu Jul 25 18:35:24 1996
@@ -100,6 +100,10 @@
#include <linux/ip_fw.h>
#include <asm/param.h>

+#ifndef HZ
+#define HZ __KERNEL_HZ
+#endif
+
#ifndef IP_FW_F_REDIR
#define IPFWADM_NO_REDIR
#define IP_FW_F_REDIR 0
@@ -1548,9 +1552,9 @@
break;
}

- sec100s = ms->expires % 100L;
- seconds = (ms->expires / 100L) % 60;
- minutes = ms->expires / 6000L;
+ sec100s = ms->expires % HZ * 100 / HZ;
+ seconds = ms->expires / HZ % 60;
+ minutes = ms->expires / (HZ * 60);

fprintf(fp, "%02ld:%02ld.%02ld ", minutes, seconds, sec100s);
