in /etc/rc.d/rc.sysinit the following code still doesn't allow
to use ARC clocks running at GMT

ARC=0
UTC=0
if [ -f /etc/sysconfig/clock ]; then
. /etc/sysconfig/clock

# convert old style clock config to new values
if [ "${CLOCKMODE}" = "GMT" ]; then
UTC=true
elif [ "${CLOCKMODE}" = "ARC" ]; then
ARC=true
fi
fi

because "${CLOCKMODE}" can't be both "ARC" and "GMT" at the same time :-(
