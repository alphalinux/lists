> what do I need to do to get modules working with a 2.0.x kernel (based
> on RedHat 4.1, i.e. ELF). If found only modutils for 2.1.x kernels and
> modutils-2.1.34 always gives me "init_module: Bad address".

Apply the following patch to your kernel headers and
rebuild modutils 2.1.34.

r~

--- linux-2.0.27/include/asm-alpha/unistd.h.orig Mon Nov 18 02:25:38 1996
+++ linux-2.0.27/include/asm-alpha/unistd.h Sat Mar 29 18:04:41 1997
@@ -214,7 +214,7 @@ type name (type1 arg1, type2 arg2, type3
type name (type1 arg1,type2 arg2,type3 arg3,type4 arg4,type5 arg5) \
{ \
extern long syscall (int, ...); \
- return syscall(__NR_##name, arg1, arg2, arg3, arg4); \
+ return syscall(__NR_##name, arg1, arg2, arg3, arg4, arg5); \
}

#endif /* __LIBRARY__ && __GNUC__ */
