#include <math.h>
#include <signal.h>

void fpe_handler(int sig) {

}

int main()
{
double a, b, c;
float d, e, f;

signal(SIGFPE, fpe_handler);

printf("about to divide 1.0 by 0.0\n");
a = 1.0;
b = 0.0;
c = a/b;
printf("result: %g\n", c);

printf("about to assign HUGE_VAL to a float\n");
/* whats the proper way to get this, anyhow? */

printf("about to print that float\n");
printf("Inf is %g\n", f);

return 0;
}
