It may be that you have to be running with glibc-1.99 and friends. I
struggled through that upgrade several weeks ago. You must fully specify
everything in your network setup, for example:

------------------

network:

NETWORKING=yes
HOSTNAME=alpha
DOMAINNAME=steve.net
GATEWAY=192.168.244.50
GATEWAYDEV=eth0
GATEWAYMASK=0.0.0.0

------------------------

ifcfg-eth0:

DEVICE=eth0
IPADDR=192.168.244.65
NETMASK=255.255.255.0
NETWORK=192.168.244.0
BROADCAST=192.168.244.255
ONBOOT=yes
BOOTP=no

---------------------------

ifcfg-lo:

DEVICE=lo
IPADDR=127.0.0.1
NETMASK=255.0.0.0
NETWORK=127.0.0.0
BROADCAST=127.255.255.255
ONBOOT=yes
BOOTP=no

--------------------

ifup patch:

*** ifup.save Tue Sep 10 12:52:20 1996
--- ifup Sun Jan 5 16:35:02 1997
***************
*** 44,50 ****
fi

ifconfig ${DEVICE} ${IPADDR} netmask ${NETMASK} broadcast ${BROADCAST}
! route add -net ${NETWORK} netmask ${NETMASK} ${DEVICE}

# this is broken! it's only here to keep compatibility with old RH sytstems
if [ "${GATEWAY}" != "" -a "${GATEWAY}" != "none" ]
--- 44,52 ----
fi

ifconfig ${DEVICE} ${IPADDR} netmask ${NETMASK} broadcast ${BROADCAST}
! # ifconfig ${DEVICE} ${IPADDR}
! route add -net ${NETWORK} netmask ${NETMASK} dev ${DEVICE}
! # route add -net ${NETWORK} ${DEVICE}

# this is broken! it's only here to keep compatibility with old RH sytstems
if [ "${GATEWAY}" != "" -a "${GATEWAY}" != "none" ]
***************
*** 57,63 ****
if [ "${GATEWAY}" != "" ]; then
if [ "${GATEWAYDEV}" = "" -o "${GATEWAYDEV}" = "${DEVICE}" ]; then
# set up default gateway
! route add default gw ${GATEWAY}
fi
fi

--- 59,65 ----
if [ "${GATEWAY}" != "" ]; then
if [ "${GATEWAYDEV}" = "" -o "${GATEWAYDEV}" = "${DEVICE}" ]; then
# set up default gateway
! route add default gw ${GATEWAY} netmask ${GATEWAYMASK} dev ${DEVICE}
fi
fi
