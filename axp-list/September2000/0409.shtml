Here's the program:

int main()
{
  double x = 1.0;
  for (x = 1.0; x != 0.0; x *=0.5)
    printf("%e\n", x);
}

Here's the end of the output:

6.324040e-322
3.162020e-322
1.581010e-322
7.905050e-323
3.952525e-323
1.976263e-323
9.881313e-324
4.940656e-324

Here's the compiler I use:

popov-141% gcc -v
Reading specs from /export/u10/gcc-2.95.1/lib/gcc-lib/alphaev6-unknown-linux-gnu/2.95.1/specs
gcc version 2.95.1 19990816 (release)

and here's the assembler (there are no trapb instructions because
this is 21264 and they aren't needed):

popov-143% cat testdenorm.s
        .file 1 "testdenorm.c"
        .set noat
        .set noreorder
        .arch ev6
.section .rodata
$LC1:
        .ascii "%e\12\0"
        .align 3
$LC0:
        .t_floating 1.00000000000000000000e0
        .align 3
$LC2:
        .t_floating 5.00000000000000000000e-1
.text
        .align 5
        .globl main
        .ent main
main:
        .eflag 48
        .frame $15,32,$26,0
        .mask 0x4008000,-32
        ldgp $29,0($27)
$main..ng:
        lda $30,-32($30)
        stq $26,0($30)
        stq $15,8($30)
        mov $30,$15
        .prologue 1
        lda $1,$LC0
        ldt $f10,0($1)
        stt $f10,16($15)
        lda $1,$LC0
        ldt $f10,0($1)
        stt $f10,16($15)
$L3:
        ldt $f10,16($15)
        cmpteqsu $f10,$f31,$f11 <= su implies ieee fp version of instruction
        fbeq $f11,$L6
        br $31,$L4
$L6:
        lda $16,$LC1
        ldt $f17,16($15)
        jsr $26,printf
        ldgp $29,0($26)
$L5:
        ldt $f10,16($15)
        lda $1,$LC2
        ldt $f11,0($1)
        multsu $f10,$f11,$f12 <= su implies ieee fp version of instruction
        fmov $f12,$f10
        stt $f10,16($15)
        br $31,$L3
$L4:
$L2:
        mov $15,$30
        ldq $26,0($30)
        ldq $15,8($30)
        lda $30,32($30)
        ret $31,($26),1
        .end main
        .ident "GCC: (GNU) 2.95.1 19990816 (release)"
